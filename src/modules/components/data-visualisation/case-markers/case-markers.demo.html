Remove heading

//Displays a single marker without a header. caseMarkerSingle = { markers: [ 'Domestic violence' ] }; //You can set heading to false to remove heading caseMarkerSingle = { markers: [ 'Domestic violence' ], heading: false }; // You can also set heading as '' providing flexibility. caseMarkerSingle = { markers: [ 'Domestic violence' ], heading: '' };

Set heading

//Displays multiple markers with a related heading. caseMarkerWithHeading = { markers: [ 'Domestic violence', 'Offender' ], heading: 'Case markers' };
